Farmer/Farm Lab Product Carbon Passport
CYPHER QUERY · NEO4J ● 
MATCH (f:Farmer)-[:OWNS]->(farm:Farm)
    -[:HAS_PLOT]->(plot:LotusPlot)
    -[:RUNS_CYCLE]->(cycle:CultivationCycle)
    -[:PRODUCES]->(batch:HarvestBatch)
    -[:HAS_CARBON]->(c:CarbonRecord)
WHERE c.co2e_result < 15.0
RETURN f.name, farm.location, c.co2e_result
ORDER BY c.co2e_result ASC
LIMIT 10
